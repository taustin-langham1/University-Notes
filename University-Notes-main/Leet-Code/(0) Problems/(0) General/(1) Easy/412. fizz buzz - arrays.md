public static void fizzBuzz (int [] Array){
    for (int i = 0; i < Array.length; i++){
        if (Array[i] % 5 == 0 && Array[i] % 3 == 0){
            System.out.println("FizzBuzz");
        }
        else if (Array[i] % 3 == 0){

            System.out.println("Fizz");
        }
        else if (Array[i] % 5 == 0){

            System.out.println("Buzz");
        }
    }
}
